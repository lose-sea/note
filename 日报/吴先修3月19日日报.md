## 题目1:  [LCR 022. 环形链表 II](https://leetcode.cn/problems/c32eOV/) 

![image-20260319215139670](/Users/lose_sea/Library/Application Support/typora-user-images/image-20260319215139670.png)

```objc
 // 哈希表
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        unordered_set<ListNode*> st; 
        ListNode* curr = head; 
        while (curr != nullptr) {
            if (st.find(curr) == st.end()) {
                st.insert(curr); 
            } else {
                return curr; 
            }
            curr = curr->next; 
        } 
        return nullptr; 
    }
}; 

// 快慢指针
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* fast = head; 
        ListNode* slow = head; 
        while (fast != nullptr && fast->next != nullptr) {
            fast = fast->next->next; 
            slow = slow->next; 
            if (fast == slow) {
                fast = head; 
                while (fast != slow) {
                    fast = fast->next; 
                    slow = slow->next; 
                } 
                return fast; 
            }
        } 
        return nullptr; 
    }
}; 
```

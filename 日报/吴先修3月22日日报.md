## 题目1:   [541. 反转字符串 II](https://leetcode.cn/problems/reverse-string-ii/)  

![image-20260322205351334](/Users/lose_sea/Library/Application Support/typora-user-images/image-20260322205351334.png)

```objc
class Solution {
public:

    void reverseString(string& s, int start, int end) { 
        int left = start; 
        int right = end; 
        while (left < right) {
            char ch = s[left]; 
            s[left] = s[right]; 
            s[right] = ch;  
            left++; 
            right--; 
        } 
    }
    string reverseStr(string s, int k) { 
        if (s.size() == 0) {
            return s; 
        }
        for (int i = 0; i < s.size(); i += (2 * k)) {
            if ((i + k) <= s.size()) { 
                reverseString(s, i, i + k - 1); 
                
            } else {
                reverseString(s, i, s.size() - 1); 
            }
        } 
        return s; 
    }
};
```
